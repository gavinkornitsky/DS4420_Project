clean_binary_target <- function(y_raw) {
  if (is.factor(y_raw)) {
    if (nlevels(y_raw) != 2) {
      stop("target must be binary")
    }
    y <- as.integer(y_raw == levels(y_raw)[2])

  } else if (is.character(y_raw)) {
    vals <- sort(unique(y_raw))
    if (length(vals) != 2) {
      stop("target must be binary")
    }
    y <- as.integer(y_raw == vals[2])

  } else if (is.logical(y_raw)) {
    y <- as.integer(y_raw)

  } else {
    y <- as.numeric(y_raw)
    u <- sort(unique(y))
    if (length(u) != 2) {
      stop("target must be binary")
    }
    if (!all(u %in% c(0, 1))) {
      y <- as.integer(y == max(u))
    }
  }

  y
}

model function definitions

# computes predicted probabilities
sigmoid <- function(z) {
  1 / (1 + exp(-z))
}

# computes bernoilli log likelihood
log_likelihood <- function(beta, X, y) {
  eta <- X %*% beta
  p <- sigmoid(eta)

  eps <- 1e-10
  p <- pmin(pmax(p, eps), 1 - eps)

  sum(y * log(p) + (1 - y) * log(1 - p))
}

# defines a normal prior on the coefficients 
log_prior <- function(beta, prior_sd = 2.5) {
  sum(dnorm(beta, mean = 0, sd = prior_sd, log = TRUE))
}

# posterior  
log_posterior <- function(beta, X, y, prior_sd = 2.5) {
  log_likelihood(beta, X, y) + log_prior(beta, prior_sd)
}

metropolis hastings sampling

run_bayes_logistic_mh <- function(
  X,
  y,
  n_iter = 20000,
  burn_in = 5000,
  step_size = 0.04,
  prior_sd = 2.5,
  seed = 4420
) {

  set.seed(seed)

  p <- ncol(X)

  beta_current <- rep(0, p)
  log_post_current <- log_posterior(beta_current, X, y, prior_sd)

  samples <- matrix(NA, nrow = n_iter, ncol = p)
  colnames(samples) <- colnames(X)

  accepted <- 0

  for (iter in 1:n_iter) {

    beta_proposal <- beta_current + rnorm(p, mean = 0, sd = step_size)

    log_post_proposal <- log_posterior(beta_proposal, X, y, prior_sd)

    log_alpha <- log_post_proposal - log_post_current

    if (log(runif(1)) < log_alpha) {
      beta_current <- beta_proposal
      log_post_current <- log_post_proposal
      accepted <- accepted + 1
    }

    samples[iter, ] <- beta_current
  }

  kept_samples <- samples[(burn_in + 1):n_iter, , drop = FALSE]

  list(
    samples = kept_samples,
    raw_samples = samples,
    acceptance_rate = accepted / n_iter
  )
}

predict_posterior_probs <- function(X_new, samples_matrix) {
  n_obs <- nrow(X_new)
  n_draws <- nrow(samples_matrix)

  prob_matrix <- matrix(NA, nrow = n_obs, ncol = n_draws)

  for (s in 1:n_draws) {
    beta_s <- samples_matrix[s, ]
    prob_matrix[, s] <- sigmoid(X_new %*% beta_s)
  }

  rowMeans(prob_matrix)
}

test_probs <- predict_posterior_probs(X_test, fit$samples)

accuracy <- mean(test_preds == y_test)

tp <- sum(test_preds == 1 & y_test == 1)
tn <- sum(test_preds == 0 & y_test == 0)

fn <- sum(test_preds == 0 & y_test == 1)

precision <- ifelse(tp + fp == 0, 0, tp / (tp + fp))
recall    <- ifelse(tp + fn == 0, 0, tp / (tp + fn))
f1_score  <- ifelse(
  precision + recall == 0,
  0,
  2 * precision * recall / (precision + recall)
)

auroc

compute_auc <- function(y_true, scores) {
  ord <- order(scores, decreasing = TRUE)
  y_true <- y_true[ord]

  pos <- sum(y_true == 1)
  neg <- sum(y_true == 0)

  if (pos == 0 || neg == 0) {
    return(NA)
  }

  tpr <- cumsum(y_true == 1) / pos
  fpr <- cumsum(y_true == 0) / neg

  tpr <- c(0, tpr, 1)
  fpr <- c(0, fpr, 1)

  auc <- sum((fpr[-1] - fpr[-length(fpr)]) *
               (tpr[-1] + tpr[-length(tpr)]) / 2)

  auc
}

auroc <- compute_auc(y_test, test_probs)
